#include <iostream>
using namespace std;

int main() {
    int arr[5] = {5, 3, 4, 1, 2};
    int i, j, temp;

    for (i = 0; i < 5 - 1; i++) {
        for (j = 0; j < 5 - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                // swap
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    cout << "Sorted array: ";
    for (i = 0; i < 5; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}
