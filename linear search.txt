#include <iostream>
using namespace std;

int main() {
    int arr[5] = {10, 20, 30, 40, 50};
    int key, i;
    bool found = false;

    cout << "Enter element to search: 40";
    cin >> key;

    for (i = 0; i < 5; i++) {
        if (arr[i] == key) {
            cout << "Element found at index " << i;
            found = true;
            break;
        }
    }

    if (!found) {
        cout << "Element not found";
    }

    return 0;
}
