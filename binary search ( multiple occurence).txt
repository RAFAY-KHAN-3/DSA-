#include <iostream>
using namespace std;

int main() {
    int arr[7] = {10, 20, 20, 20, 30, 40, 50};
    int key;
    int low = 0, high = 6, mid, index = -1;

    cout << "Enter element to search: ";
    cin >> key;

    // Binary search to find one occurrence
    while (low <= high) {
        mid = (low + high) / 2;

        if (arr[mid] == key) {
            index = mid;
            break;
        }
        else if (arr[mid] < key) {
            low = mid + 1;
        }
        else {
            high = mid - 1;
        }
    }

    if (index == -1) {
        cout << "Element not found";
    }
    else {
        cout << "Element found at indices: ";

        // Left side
        int i = index;
        while (i >= 0 && arr[i] == key) {
            i--;
        }

        // Right side
        i++;
        while (i < 7 && arr[i] == key) {
            cout << i << " ";
            i++;
        }
    }

    return 0;
}
